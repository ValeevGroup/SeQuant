     SEQUANT_ASSERT(std::none_of(str.begin(), str.end(), [](wchar_t c) {
       return c > std::numeric_limits<char16_t>::max();
     }));

   }

     static_assert(sizeof(wchar_t) >= sizeof(char16_t));
   
     return std::wstring(stream.begin(), stream.end());
   }
