   template <typename X>
   X numeric_cast(const sequant::rational &r) {
     if constexpr (std::is_integral_v<X>) {
       SEQUANT_ASSERT(denominator(r) == 1);
       return boost::numeric_cast<X>(numerator(r));
     } else {
       return boost::numeric_cast<X>(numerator(r)) /
              boost::numeric_cast<X>(denominator(r));
     }
   };
